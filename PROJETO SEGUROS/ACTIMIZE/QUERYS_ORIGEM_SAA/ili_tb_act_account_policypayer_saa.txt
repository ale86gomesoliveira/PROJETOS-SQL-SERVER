	     TRIM(REPLACE(REPLACE(REPLACE(REPLACE(C.NM_RAZAOSOCIAL,'"',''),'|',''),'*',''),'#','')) IS NULL
	    THEN TRIM(REPLACE(REPLACE(REPLACE(REPLACE(C.NM_PRIMNOMECLIENTE,'"',''),'|',''),'*',''),'#','')) || 
	     TRIM(REPLACE(REPLACE(REPLACE(REPLACE(C.NM_NOMEMEIOCLIENTE,'"',''),'|',''),'*',''),'#','')) || 
	     TRIM(REPLACE(REPLACE(REPLACE(REPLACE(C.NM_ULTNOMECLIENTE,'"',''),'|',''),'*',''),'#',''))

'' AS PARTY_ALIAS_MAIDEN_NAME3_NM,
'' AS DOCUMENT_KEY1_ID,

(CASE 
    WHEN C.TP_NATJUR = 'PF' THEN 'CPF' 
    WHEN C.TP_NATJUR = 'PJ' THEN 'CNPJ'
    ELSE '' 
END) AS DOCUMENT_TYPE1_CD,
'' AS ISSUE_COUNTRY1_CD,
'' AS ISSUE_STATE_PROVINCE1_CD,
'' AS DOCUMENT_KEY2_ID,
'' AS DOCUMENT_NUMBER2_NR,
'' AS DOCUMENT_TYPE2_CD,
'' AS ISSUE_COUNTRY2_CD,

'' AS EMAIL_RELATION_TYPE1_CD,
(SELECT DS_CONTATOCLIENTE 
    FROM SAA.CLIENTECONTATO E
    WHERE E.CD_CLIENTE = C.CD_CLIENTE AND E.TP_CONTATOCLIENTE = 'EM') AS PARTY_EMAIL_ADDRESS1_NM,
'' AS EMAIL_RELATION_TYPE2_CD,
'' AS PARTY_EMAIL_ADDRESS2_NM,
'' AS PHONE_RELATION_TYPE1_CD,
(SELECT DS_CONTATOCLIENTE 
    FROM SAA.CLIENTECONTATO E
    WHERE E.CD_CLIENTE = C.CD_CLIENTE AND E.TP_CONTATOCLIENTE = 'HO') AS PARTY_PHONE1_NR,
'' AS PHONE_RELATION_TYPE2_CD,
(SELECT DS_CONTATOCLIENTE 
    FROM SAA.CLIENTECONTATO E
    WHERE E.CD_CLIENTE = C.CD_CLIENTE AND E.TP_CONTATOCLIENTE = 'CP') AS PARTY_PHONE2_NR,
'' AS PHONE_RELATION_TYPE3_CD,
(SELECT DS_CONTATOCLIENTE 
    FROM SAA.CLIENTECONTATO E
    WHERE E.CD_CLIENTE = C.CD_CLIENTE AND E.TP_CONTATOCLIENTE = 'BU') AS PARTY_PHONE3_NR,

'' AS STATE_PROVINCE3_CD,
'' AS STREET3_NM,

INNER JOIN SAA.APOLCLIENTE B ON A.NR_APOLICE = B.NR_APOLICE
INNER JOIN SAA.CLIENTE C ON A.CD_PROPONENTE = C.CD_CLIENTE 
INNER JOIN SAA.CLIENTEENDERECO D ON C.CD_CLIENTE = D.CD_CLIENTE
INNER JOIN SAA.CLIENTECONTATO E ON C.CD_CLIENTE = E.CD_CLIENTE
INNER JOIN SAA.PROFISSAO F ON C.CD_PROFISSAO = F.CD_PROFISSAO
WHERE D.TP_ENDERECOCLIENTE = 'PR' 
AND B.TP_ENDERECOCLIENTE = 'PR' 
AND A.CD_PROPONENTE = C.CD_CLIENTE
AND E.TP_CONTATOCLIENTE = 'EM'
AND (A.CD_STATUSAPOLICE = '1'OR A.CD_STATUSAPOLICE = '2'OR A.CD_STATUSAPOLICE = '3'OR A.CD_STATUSAPOLICE = '4'OR A.CD_STATUSAPOLICE = 'Y')
AND ROWNUM <= 200
