WITH MAX_DT_SIST
AS
(SELECT 
    DISTINCT
	TPR.CD_CONTROLE,
	TPR.CD_CONTROLE_TITULAR,
	TPR.CD_VERIFICADOR,
	TPR.CD_RELACAO,
	TPR.DT_NASCIMENTO, 
	MAX(DT_SISTEMA) AS dt_sistema 
	FROM DBTCTO.TB_PEP_RESULTADOS TPR
	GROUP BY 
	TPR.CD_CONTROLE, 
	CD_CONTROLE_TITULAR,
	TPR.CD_VERIFICADOR,
	TPR.CD_RELACAO,
	TPR.DT_NASCIMENTO
	)
SELECT 
DISTINCT
(TPR.CD_CONTROLE CONCAT TPR.CD_VERIFICADOR) AS Control_id,
CD_FILIAL AS Branch_cd,
TPR.CD_VERIFICADOR AS Verification_Cd,
max(TPR.DT_SISTEMA) AS System_Date_Dt,
CASE 
	WHEN TPR.CD_RELACAO = '1' THEN 'FUNCIONÁRIO'
	WHEN TPR.CD_RELACAO = '2'  THEN 'CLIENTE'
	WHEN TPR.CD_RELACAO = '3'  THEN 'FRANQUEADO'
	WHEN TPR.CD_RELACAO = '4'  THEN 'PRESTADOR'
	WHEN TPR.CD_RELACAO = '5'  THEN 'CANDI FRANQUE'
	WHEN TPR.CD_RELACAO = '6'  THEN 'CANDI FUNCIO'	
	WHEN TPR.CD_RELACAO = '7'  THEN 'EVENTUAL'
	WHEN TPR.CD_RELACAO = '8'  THEN 'SÓCIO FRANQUEADO'
	WHEN TPR.CD_RELACAO = '9'  THEN 'PRESTADOR KYOEI'
	WHEN TPR.CD_RELACAO = '10' THEN 'CLIENTE KYOEI'
	WHEN TPR.CD_RELACAO = '11' THEN 'FUNCIONARIO KYOEI'
	WHEN TPR.CD_RELACAO = '12' THEN 'CLIENTE VG'
	WHEN TPR.CD_RELACAO = '13' THEN 'CORRETOR VG'
	WHEN TPR.CD_RELACAO = '14' THEN 'PRESTADOR VG'
	WHEN TPR.CD_RELACAO = '15' THEN 'FAVORECIDO'
	WHEN TPR.CD_RELACAO = '16' THEN 'SEGURADO VG'
	WHEN TPR.CD_RELACAO = '17' THEN 'BENEFICIARIO VG'
	WHEN TPR.CD_RELACAO = '18' THEN 'SUBESTIPULANTE VG'
	WHEN TPR.CD_RELACAO = '19' THEN 'POTENCIAL CLIENTE'
	WHEN TPR.CD_RELACAO = '20' THEN 'teste'
	WHEN TPR.CD_RELACAO = '21' THEN 'Teste'
	WHEN TPR.CD_RELACAO = '22' THEN 'FAVORE MASSIFICA'
END Relationship_tx,
TPR.CD_CONTROLE_TITULAR AS Holder_Control_Cd,
CD_FILIAL_TITULAR AS Holder_Branch_Cd,
CD_VERIFICADOR_TITULAR AS Holder_Verification_Cd,
CD_NOSSO_NUMERO_TITULAR AS Holder_Number_Cd,
DS_NOME AS Name_Nm,
TPR.DT_NASCIMENTO AS Birth_date_Dt,
CD_SEXO AS Gender_Cd,
DS_RELACIONAMENTO AS Relationship_desc_tx
FROM DBTCTO.TB_PEP_RESULTADOS tpr 
INNER JOIN MAX_DT_SIST MDS ON TPR.DT_SISTEMA = MDS.DT_SISTEMA
WHERE tPR.CD_CONTROLE_TITULAR IN ('053773726','054511276','000016086','045294826')
AND TPR.CD_CONTROLE_TITULAR = '054511276'
GROUP BY
TPR.CD_CONTROLE,
CD_FILIAL,
TPR.CD_VERIFICADOR,
TPR.CD_RELACAO,
TPR.CD_CONTROLE_TITULAR,
CD_FILIAL_TITULAR,
CD_VERIFICADOR_TITULAR,
CD_NOSSO_NUMERO_TITULAR,
DS_NOME,
TPR.DT_NASCIMENTO,
CD_SEXO,
DS_RELACIONAMENTO
