SELECT
DISTINCT 
(CD_CONTROLE CONCAT CD_FILIAL CONCAT CD_VERIFICADOR) AS Control_Id, 
--COUNT(DISTINCT CD_CONTROLE) AS CD_CONTROLE,
CD_FILIAL AS Branch_Cd,
CD_VERIFICADOR AS Verification_Cd,
trim(CAST(DS_NOME AS nvarchar(500))) AS Person_Name_Nm, 
DT_NASCIMENTO AS Person_Date_Birth_Dt,
TRIM(DS_PAIS_ORIGEM) AS Country_Origin_nm,
IN_PAIS_SANCIONADO AS Sanctioned_Country_Nm,
CD_RAMO_ATIVIDADE AS business_line_cd
FROM DBTCTO.TB_PEP_PESSOAS tp
